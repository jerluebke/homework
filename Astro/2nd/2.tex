\section*{Task 2}

\begin{enumerate}[label=\textbf{\large(\alph*)}, itemsep=2\baselineskip]

\item
    Estimated gained frequency by a photon with $\nu_0=\SI{282}{\giga\hertz}$ after
    being scattered by a high energy electron:
    \begin{align*}
        \Eph' &= \Eph \left(1+\frac{kT}{m_e c^2}\right) \\
        \implies \Delta\nu &= \nu'-\nu_0 = \frac{kT}{m_e c^2}\nu_0
    \end{align*}

    For a plasma with $n_e = \SI{3e3}{\metre\cubed}$, $T_e=10^6\si{\kelvin}$,
    the first increase in frequency is:
    \begin{equation*}
        \nu'\approx\SI{282.048e9}{\hertz}
    \end{equation*}

\end{enumerate}
